N<-100
bmu<-numeric(1000)
#給他1000次的空間，不用也不會錯因為他允許如此，但一定要隨意給他一個初始職否則會發師錯誤
metasample<-rnorm(N,5,2)
for(i in 1:1000){
  rndno<-as.integer(runif(N)*N)+1
  smp<-metasample[rndno]#抽一次
  bmu[i]<-mean(smp)#第?次
}
#上面100會是常態分佈
p<-density(bmu)#要畫圖了
plot(p)
